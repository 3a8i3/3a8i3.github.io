Datos JSON para prácticas

Ejemplo 

[{ 

	"titulo": "String", 

	"autor": { 

		"nombre": "String", 

		"nacimiento": { 

			"fecha": "String", 

			"ciudad": "String", 

			"pais": "String", 

		}, 

		biografia: "String", 

		"foto": "url" 

	}, 

	"anho": "String", 

	"editorial": "String", 

	"temas": "String", 

	"resumen": "String", 

	"portada": "url" 

},  

{ ... } ]
